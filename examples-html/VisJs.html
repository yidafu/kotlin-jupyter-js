fun custom(x: Double, y: Double): Double {
  return Math.sin(x / 50) * Math.cos(y / 50) * 50 + 50;
}
  
val dataList = mutableListOf<Map<String, Double>>()

val steps = 50; // number of datapoints will be steps*steps
val axisMax = 314;
val axisStep = axisMax / steps;
var x = 0.0;
var counter: Double = 0.0;
while (x < axisMax) {
    var y = 0.0;
    while (y < axisMax) {
        val v = custom(x, y);
        counter += 1.0;
        dataList.add(mapOf( "id" to counter,  "x" to x, "y" to y, "z" to v, "style" to v ))
        y += axisStep;
    }
    x += axisStep;
}
